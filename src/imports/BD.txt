BASE DE DATOS DEL PROYECTO

Este archivo contiene la estructura definitiva de la base de datos MySQL del sistema de Seguimiento de Egresados y Bolsa Laboral.

Utiliza esta estructura únicamente como referencia funcional para validar y ajustar el diseño existente.

No modifiques el modelo de datos.

No asumas tablas, campos o relaciones que no estén definidos aquí.


CREATE DATABASE seg_egresado_bolsa;
USE seg_egresado_bolsa;

CREATE TABLE usuario (
    id_usuario INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    nombre_usuario VARCHAR(50) NOT NULL,
    password VARCHAR(255) NOT NULL,
    correo VARCHAR(100) NOT NULL,
    estado_usuario VARCHAR(20) NOT NULL,
    fecha_creacion DATETIME NOT NULL,
    ultimo_acceso DATETIME,

    CONSTRAINT uq_nombre_usuario UNIQUE(nombre_usuario),
    CONSTRAINT uq_correo UNIQUE(correo)
);

CREATE TABLE administrador (
    id_usuario INT PRIMARY KEY NOT NULL,

    FOREIGN KEY (id_usuario)
    REFERENCES usuario(id_usuario)
);

CREATE TABLE facultad (
    id_facultad INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    nombre_facultad VARCHAR(100) NOT NULL,
    siglas VARCHAR(10) NOT NULL,
    descripcion TEXT,
    estado_facultad VARCHAR(20) NOT NULL,

    CONSTRAINT uq_nombre_facultad UNIQUE(nombre_facultad),
    CONSTRAINT uq_siglas_facultad UNIQUE(siglas)
);

CREATE TABLE carrera (
    id_carrera INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    nombre_carrera VARCHAR(100) NOT NULL,
    grado_academico VARCHAR(50) NOT NULL,
    duracion_años INT NOT NULL,
    estado_carrera VARCHAR(20) NOT NULL,
    id_facultad INT NOT NULL,

    CONSTRAINT uq_nombre_carrera UNIQUE(nombre_carrera),

    FOREIGN KEY (id_facultad)
    REFERENCES facultad(id_facultad)
);

CREATE TABLE egresado (
    id_usuario INT PRIMARY KEY NOT NULL,
    dni CHAR(8) NOT NULL,
    nombre_egresado VARCHAR(100) NOT NULL,
    apellidos_egresado VARCHAR(100) NOT NULL,
    telefono VARCHAR(15),
    direccion VARCHAR(255),
    fecha_egreso DATE NOT NULL,
    sexo CHAR(1) NOT NULL,
    id_carrera INT NOT NULL,

    CONSTRAINT uq_dni UNIQUE(dni),

    CONSTRAINT uq_nombre_completo
    UNIQUE(nombre_egresado, apellidos_egresado),

    FOREIGN KEY (id_usuario)
    REFERENCES usuario(id_usuario),

    FOREIGN KEY (id_carrera)
    REFERENCES carrera(id_carrera)
);

CREATE TABLE empresa (
    id_usuario INT PRIMARY KEY NOT NULL,
    ruc CHAR(11) NOT NULL,
    razon_social VARCHAR(150) NOT NULL,
    nombre_comercial VARCHAR(150),
    sector VARCHAR(100) NOT NULL,
    direccion VARCHAR(255) NOT NULL,
    telefono VARCHAR(15),
    pagina_web VARCHAR(100),

    CONSTRAINT uq_ruc UNIQUE(ruc),
    CONSTRAINT uq_razon_social UNIQUE(razon_social),

    FOREIGN KEY (id_usuario)
    REFERENCES usuario(id_usuario)
);

CREATE TABLE historial_laboral (
    id_historial INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    nombre_empresa VARCHAR(150) NOT NULL,
    cargo VARCHAR(100) NOT NULL,
    fecha_inicio DATE NOT NULL,
    fecha_fin DATE,
    salario DECIMAL(10,2),
    modalidad VARCHAR(50) NOT NULL,
    actual BOOLEAN NOT NULL,
    id_egresado INT NOT NULL,

    FOREIGN KEY (id_egresado)
    REFERENCES egresado(id_usuario)
);

CREATE TABLE encuesta_seguimiento (
    id_encuesta INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    fecha_registro DATE NOT NULL,
    estado_laboral VARCHAR(50) NOT NULL,
    nombre_empresa_actual VARCHAR(150),
    cargo_actual VARCHAR(100),
    area_trabajo VARCHAR(100),
    sueldo_mensual DECIMAL(10,2),
    tipo_contrato VARCHAR(50),
    satisfaccion_profesional VARCHAR(50),
    tiempo_conseguir_empleo VARCHAR(50),
    observaciones TEXT
);

CREATE TABLE seguimiento_egresado (
    id_seguimiento INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    id_encuesta INT NOT NULL,
    id_egresado INT NOT NULL,
    fecha_asociacion DATETIME NOT NULL,
    
    CONSTRAINT uq_encuesta_egresado
    UNIQUE(id_encuesta, id_egresado),

    FOREIGN KEY (id_encuesta)
    REFERENCES encuesta_seguimiento(id_encuesta),

    FOREIGN KEY (id_egresado)
    REFERENCES egresado(id_usuario)
);

CREATE TABLE oferta_laboral (
    id_oferta INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    titulo VARCHAR(150) NOT NULL,
    descripcion TEXT NOT NULL,
    puesto VARCHAR(100) NOT NULL,
    area VARCHAR(100) NOT NULL,
    ubicacion VARCHAR(150) NOT NULL,
    modalidad VARCHAR(50) NOT NULL,
    tipo_contrato VARCHAR(50) NOT NULL,
    salario DECIMAL(10,2),
    requisitos TEXT,
    fecha_publicacion DATE NOT NULL,
    fecha_cierre DATE NOT NULL,
    estado_oferta VARCHAR(20) NOT NULL,
    id_empresa INT NOT NULL,

    FOREIGN KEY (id_empresa)
    REFERENCES empresa(id_usuario)
);

CREATE TABLE postulacion (
    id_postulacion INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    id_egresado INT NOT NULL,
    id_oferta INT NOT NULL,
    fecha_postulacion DATETIME NOT NULL,
    estado_postulacion VARCHAR(20) NOT NULL,
    observaciones TEXT,
    cv_adjunto VARCHAR(255),

    CONSTRAINT uq_postulacion
    UNIQUE(id_egresado, id_oferta),

    FOREIGN KEY (id_egresado)
    REFERENCES egresado(id_usuario),

    FOREIGN KEY (id_oferta)
    REFERENCES oferta_laboral(id_oferta)
);

CREATE TABLE auditoria (
    id_auditoria INT PRIMARY KEY AUTO_INCREMENT,
    tabla_afectada VARCHAR(50) NOT NULL,
    accion VARCHAR(20) NOT NULL,
    id_registro INT NOT NULL,
    descripcion TEXT,
    fecha_evento DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuario_bd VARCHAR(100)
);

CREATE TABLE notificacion (
    id_notificacion INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario INT NOT NULL,
    titulo VARCHAR(150) NOT NULL,
    mensaje TEXT NOT NULL,
    leido BOOLEAN DEFAULT FALSE,
    fecha_envio DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_usuario)
    REFERENCES usuario(id_usuario)
);

CREATE TABLE recuperacion_password (
    id_recuperacion INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario INT NOT NULL,
    token VARCHAR(255) NOT NULL,
    fecha_expiracion DATETIME NOT NULL,

    FOREIGN KEY (id_usuario)
    REFERENCES usuario(id_usuario)
);
